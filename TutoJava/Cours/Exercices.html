Exercice 2 - Calculer la somme de deux nombres

Écrivez une fonction qui prend deux nombres en entrée et retourne leur somme.


function somme(a, b) {
    return a + b;
}
console.log(somme(5, 3)); // Résultat attendu : 8
Exercice 3 - Vérifier si un nombre est pair ou impair

Écrivez une fonction qui prend un nombre en entrée et vérifie s'il est pair ou impair.


function estPair(n) {
    return n % 2 === 0 ? "Pair" : "Impair";
}
console.log(estPair(4)); // Résultat attendu : "Pair"
console.log(estPair(7)); // Résultat attendu : "Impair"

Exercice 5 - Inverser un tableau

Écrivez une fonction qui inverse l'ordre des éléments d'un tableau.


function inverserTableau(arr) {
    return arr.reverse();
}
console.log(inverserTableau([1, 2, 3, 4])); // Résultat attendu : [4, 3, 2, 1]
Exercice 6 - Calculer la factorielle d'un nombre

Écrivez une fonction qui calcule la factorielle d'un nombre.


function factorielle(n) {
    let resultat = 1;
    for (let i = 1; i <= n; i++) {
        resultat *= i;
    }
    return resultat;
}
console.log(factorielle(5)); // Résultat attendu : 120
